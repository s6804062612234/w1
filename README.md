#include <stdio.h>


int main(void)
{
	float mid, final, q1, q2, lab, sit, max1, max2, max3, max4, max5, max6, per1, per2, per3, per4, per5, per6, sum;
	
	max1 = 100; max2 = 120; max3 = 60; max4 = 60; max5 = 16; max6 = 16;
	per1 = 30; per2 = 35; per3 = 12; per4 = 13; per5 = 5; per6 = 5;
	
	scanf("%f %f %f %f %f %f", &mid, &final, &q1, &q2, &lab, &sit);
	
	mid		= mid 	* per1 / max1;
	final	= final * per2 / max2;
	q1 		= q1 	* per3 / max3;
	q2 		= q2 	* per4 / max4;
	lab 	= lab 	* per5 / max5;
	sit 	= sit 	* per6 / max6;
	sum 	= mid + final + q1 + q2 + lab + sit;
	
	
	//display score
	printf("%10s = %8f\n", "Mid", mid);
	printf("%10s = %8f\n", "Final", final);
	printf("%10s = %8f\n", "Q1", q2);
	printf("%10s = %8f\n", "Q2", q2);
	printf("%10s = %8f\n", "Lab", lab);
	printf("%10s = %8f\n", "Sit", sit);
	printf("***%f***", sum);
}
